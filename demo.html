Add options through JS and show more options

</dropdown-js>

dropdownJsInitFunctions.push(function() {
  var demo3 = document.getElementById('demo-3');
  for(var i=1; i<=99; i++) {
    demo3.addOption(i, 'Age of ' + i);
  }
});

Remove options through JS
<dropdown-js id="demo-4">
  <option value="#0000ff">Blue</option>
  <option value="#00ffff">Cyan</option>
  <option value="#00ff00">Green</option>
  <option value="#ff00ff">Magenta</option>
  <option value="#ff0000">Red</option>

</dropdown-js>

dropdownJsInitFunctions.push(function() {
  var demo4 = document.getElementById('demo-4');
  demo4.selectOption('#00ffff');

  demo4.removeOption('#00ffff');
  demo4.removeOption('#ff00ff');
  demo4.removeOption('#ffff00');
});
